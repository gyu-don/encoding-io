pub struct TextReader<R> { /* fields omitted */ }

TextReader is a wrapper for a non-utf8 encoded text reader. It behaves as a utf8 encoded text reader.

Examples:

use std::fs::File;
use std::io::{Read, BufRead};
use encoding_io::TextReader;
use encoding_rs::*;
let mut reader = TextReader::new(
        File::open("shiftjis-text.txt").expect("failed to open."),
        SHIFT_JIS);
for line in reader.lines() {
    let line = line.expect("Failed to read.");
    println!("{}", line);
}
